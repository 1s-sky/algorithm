public static String solution(int n, int[] nums){
    StringBuilder answer = new StringBuilder();
    for(int i=0; i<n; i++){
        if(i==0) answer.append(nums[i]).append(' ');
        else{
            if(nums[i] > nums[i-1]) answer.append(nums[i]).append(' ');
        }
    }
    return answer.toString();
}
